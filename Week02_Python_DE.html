Upsert: PostgreSQL vs Oracle SQL

sql/postgres_upsert.sql
-- PostgreSQL
INSERT INTO dim_customer (customer_id, city, updated_at)
VALUES (:customer_id, :city, NOW())
ON CONFLICT (customer_id)
DO UPDATE SET
  city = EXCLUDED.city,
  updated_at = NOW();
sql/oracle_merge.sql
-- Oracle SQL
MERGE INTO dim_customer d
USING (
  SELECT :customer_id customer_id, :city city FROM dual
) s
ON (d.customer_id = s.customer_id)
WHEN MATCHED THEN UPDATE SET
  d.city = s.city,
  d.updated_at = SYSTIMESTAMP
WHEN NOT MATCHED THEN INSERT (customer_id, city, updated_at)
VALUES (s.customer_id, s.city, SYSTIMESTAMP);
📌 PostgreSQL tối ưu với ON CONFLICT, còn Oracle chuẩn production thường dùng MERGE cho upsert.

Pagination + retry backoff

python/extract_api.py
import time
import requests

def fetch_orders(base_url, token):
    page = 1
    all_rows = []
    while True:
        resp = requests.get(
            f"{base_url}/orders",
            headers={"Authorization": f"Bearer {token}"},
            params={"page": page, "page_size": 500},
            timeout=30,
        )
        if resp.status_code in [429, 500, 502, 503, 504]:
            time.sleep(min(2 ** page, 30))
            continue
        resp.raise_for_status()
        payload = resp.json()
        all_rows.extend(payload["data"])
        if not payload.get("has_more"):
            break
        page += 1
    return all_rows
✅ Luôn đặt timeout cho request và chỉ retry các mã lỗi tạm thời. Tránh retry vô hạn gây treo pipeline.

Structured logging + retry decorator

python/logging_retry.py
import json
import logging
import time

logger = logging.getLogger("etl")
logger.setLevel(logging.INFO)
handler = logging.FileHandler("etl.log", encoding="utf-8")
logger.addHandler(handler)

def log_event(level, **kwargs):
    message = json.dumps(kwargs, ensure_ascii=False)
    logger.log(level, message)

def retry(max_attempts=3, delay=2):
    def decorator(func):
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    log_event(logging.WARNING, event="retry", attempt=attempt, error=str(e))
                    if attempt == max_attempts:
                        raise
                    time.sleep(delay * attempt)
        return wrapper
    return decorator
⚠️ Tránh except: pass. Điều này làm mất tín hiệu lỗi và khiến pipeline fail ngầm trong production.

python/pipeline.py
import argparse
import os
from sqlalchemy import create_engine, text

class ETLPipeline:
    def __init__(self, engine, dialect):
        self.engine = engine
        self.dialect = dialect

    def extract(self, run_date):
        return [{"order_id": 1, "city": "HCM", "run_date": run_date}]

    def transform(self, rows):
        return rows

    def load(self, rows):
        pg_sql = "INSERT INTO stg_orders(order_id, city, run_date) VALUES (:order_id,:city,:run_date) ON CONFLICT (order_id) DO UPDATE SET city=EXCLUDED.city, run_date=EXCLUDED.run_date"
        ora_sql = "MERGE INTO stg_orders t USING (SELECT :order_id order_id, :city city, :run_date run_date FROM dual) s ON (t.order_id=s.order_id) WHEN MATCHED THEN UPDATE SET t.city=s.city, t.run_date=s.run_date WHEN NOT MATCHED THEN INSERT (order_id,city,run_date) VALUES (s.order_id,s.city,s.run_date)"
        sql = pg_sql if self.dialect == "postgres" else ora_sql
        with self.engine.begin() as conn:
            for row in rows:
                conn.execute(text(sql), row)

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--date", required=True)
    parser.add_argument("--dialect", choices=["postgres", "oracle"], default="postgres")
    args = parser.parse_args()

    pg_dsn = os.getenv("PG_DSN")
    oracle_dsn = os.getenv("ORACLE_DSN")
    url = pg_dsn if args.dialect == "postgres" else oracle_dsn
    pipeline = ETLPipeline(create_engine(url), args.dialect)
    pipeline.load(pipeline.transform(pipeline.extract(args.date)))
