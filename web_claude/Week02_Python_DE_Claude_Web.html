🐍 Python trong vai trò Data Engineer

Python là ngôn ngữ "keo dán" của DE. Bạn không dùng Python để viết web hay machine learning — bạn dùng nó để orchestrate dữ liệu: kéo từ nguồn, biến đổi, đẩy vào destination.

  • Extract: đọc CSV, JSON, Parquet; gọi REST API; query database
  • Transform: clean null, merge bảng, aggregate, validate schema
  • Load: ghi vào PostgreSQL, Parquet, S3 — incremental hoặc full load
  • Orchestrate: chạy theo thứ tự, retry khi fail, log mọi bước

🎯 Sau tuần này bạn có thể

  • Đọc và ghi mọi format file phổ biến (CSV, JSON, Parquet) bằng pandas
  • Kết nối PostgreSQL từ Python, thực hiện bulk insert và incremental load an toàn
  • Gọi REST API có pagination và xử lý rate limit không bị block
  • Viết code có error handling chuẩn, không crash khi gặp dữ liệu xấu
  • Tổ chức ETL thành module có thể tái sử dụng, test được
  • Commit artifact week2_python_etl/ hoàn chỉnh lên GitHub
📌 Tư duy quan trọng nhất tuần này: Code phải chạy được ở lần thứ 2, 3, 10 với cùng kết quả. Đây là nguyên tắc idempotency — tư duy nền tảng của mọi DE production.

import pandas as pd
import logging

logger = logging.getLogger(__name__)

def read_csv_safe(filepath: str) -> pd.DataFrame:
    """Đọc CSV với schema check và null report."""
    try:
        df = pd.read_csv(
            filepath,
            dtype_backend="numpy_nullable",  # handle NA tốt hơn
            parse_dates=["created_at", "updated_at"]
        )
        logger.info(f"Đọc {filepath}: {df.shape[0]} rows, {df.shape[1]} cols")

        # Báo cáo null ngay sau khi đọc
        null_report = df.isna().sum()
        null_cols = null_report[null_report > 0]
        if not null_cols.empty:
            logger.warning(f"Columns có null: {null_cols.to_dict()}")

        return df
    except FileNotFoundError:
        logger.error(f"File không tồn tại: {filepath}")
        raise

import glob

def read_daily_files(pattern: str) -> pd.DataFrame:
    """Đọc tất cả file khớp pattern, ghép lại thành 1 DataFrame."""
    files = glob.glob(pattern)
    if not files:
        raise ValueError(f"Không tìm thấy file nào khớp: {pattern}")
    logger.info(f"Tìm thấy {len(files)} file: {files}")
    dfs = [pd.read_csv(f) for f in files]
    combined = pd.concat(dfs, ignore_index=True)
    logger.info(f"Sau concat: {combined.shape[0]} rows tổng")
    return combined

🧹 Xử lý null — khi nào drop, khi nào fill

def clean_orders(df: pd.DataFrame) -> pd.DataFrame:
    """Làm sạch bảng orders với strategy rõ ràng."""
    original_len = len(df)

    # Drop row nếu thiếu key business columns
    df = df.dropna(subset=["order_id", "user_id", "amount"])
    logger.info(f"Dropped {original_len - len(df)} rows thiếu key columns")

    # Fill null cho optional columns với giá trị mặc định
    df["discount_amount"] = df["discount_amount"].fillna(0.0)
    df["notes"] = df["notes"].fillna("")
    df["channel"] = df["channel"].fillna("unknown")

    # Normalize dtype
    df["amount"] = df["amount"].astype(float)
    df["order_id"] = df["order_id"].astype(str)
    df["created_at"] = pd.to_datetime(df["created_at"])

    return df

def enrich_orders(orders: pd.DataFrame, users: pd.DataFrame) -> pd.DataFrame:
    """Enrich orders với user info. Left join để giữ tất cả orders."""
    before = len(orders)

    merged = pd.merge(
        orders,
        users[["user_id", "email", "country", "tier"]],
        on="user_id",
        how="left"  # giữ tất cả orders dù không tìm thấy user
    )

    after = len(merged)
    if after != before:
        # Nếu count thay đổi → có duplicate key trong users
        logger.warning(f"Row count thay đổi sau merge: {before} → {after}")

    # Kiểm tra unmatched orders
    unmatched = merged[merged["email"].isna()].shape[0]
    if unmatched > 0:
        logger.warning(f"{unmatched} orders không tìm thấy user tương ứng")

    return merged

def create_daily_summary(orders: pd.DataFrame) -> pd.DataFrame:
    """Tổng hợp orders theo ngày và channel."""
    orders["date"] = orders["created_at"].dt.date

    summary = orders.groupby(["date", "channel", "country"]).agg(

import pandas as pd
import logging

logger = logging.getLogger(__name__)

def clean_orders(filepath: str) -> pd.DataFrame:
    df = pd.read_csv(filepath)
    logger.info(f"Đọc {len(df)} rows từ {filepath}")

    # Null report
    nulls = df.isna().sum()
    for col, cnt in nulls[nulls > 0].items():
        logger.warning(f"  {col}: {cnt} nulls")

    before = len(df)
    df = df.dropna(subset=["order_id", "amount"])
    logger.info(f"Dropped {before - len(df)} rows")

    df["discount_amount"] = df["discount_amount"].fillna(0.0)
    return df

def read_orders_from_db(engine, start_date: str, end_date: str) -> pd.DataFrame:
    """Đọc orders trong khoảng thời gian. Parameterized để tránh injection."""
    query = text("""
        SELECT order_id, user_id, amount, channel, created_at
        FROM orders
        WHERE created_at >= :start_date
          AND created_at < :end_date
        ORDER BY created_at
    """)

        df = pd.read_sql_query(
            query,
            conn,
            params={"start_date": start_date, "end_date": end_date}
        )
    logger.info(f"Đọc {len(df)} orders từ {start_date} đến {end_date}")
    return df

def load_to_db(df: pd.DataFrame, engine, table_name: str, if_exists: str = "append"):
    """
    if_exists options:
    - "append": thêm vào cuối — dùng cho incremental load
    - "replace": xóa bảng cũ và tạo lại — dùng cho full refresh
    - "fail": báo lỗi nếu bảng đã tồn tại — dùng để bảo vệ data
    """
    df.to_sql(

def incremental_load(df: pd.DataFrame, engine, table_name: str, key_col: str):
    """Chỉ insert rows có key_col chưa tồn tại trong DB."""
    # Lấy danh sách keys đã có trong DB
    existing_query = text(f"SELECT {key_col} FROM {table_name}")
    with engine.connect() as conn:
        existing_keys = pd.read_sql_query(existing_query, conn)[key_col].tolist()

    # Lọc chỉ rows chưa có
    new_rows = df[~df[key_col].isin(existing_keys)]

    logger.info(f"Tổng: {len(df)} rows · Đã có: {len(existing_keys)} · Mới: {len(new_rows)}")

    if len(new_rows) > 0:
        new_rows.to_sql(table_name, engine, if_exists="append", index=False)
        logger.info(f"Inserted {len(new_rows)} rows mới")
    else:
        logger.info("Không có rows mới để insert")

    return len(new_rows)

extract.py · rate limit handler
import time
import random

def request_with_retry(session, url: str, params: dict = None,
                        max_retries: int = 5) -> dict:
    """Gọi API với exponential backoff khi gặp 429 hoặc 5xx."""
    for attempt in range(max_retries):
        try:
            response = session.get(url, params=params, timeout=30)

            if response.status_code == 429:
                # Rate limit hit — đọc Retry-After header nếu có
                retry_after = int(response.headers.get("Retry-After", 60))
                logger.warning(f"Rate limited. Chờ {retry_after}s...")
                time.sleep(retry_after)
                continue

            response.raise_for_status()
            return response.json()

        except requests.exceptions.RequestException as e:
            if attempt == max_retries - 1:
                raise
            # Exponential backoff: 1s, 2s, 4s, 8s, 16s + jitter
            wait = (2 ** attempt) + random.uniform(0, 1)
            logger.warning(f"Attempt {attempt+1} failed: {e}. Retry sau {wait:.1f}s")
            time.sleep(wait)
✅ Jitter (random delay): Thêm random.uniform(0,1) vào wait time để tránh "thundering herd" — khi nhiều client cùng retry đúng một lúc sau đó làm server càng quá tải hơn.

🔁 Retry Decorator tái sử dụng

import functools
import time
import logging

logger = logging.getLogger(__name__)

def retry(max_attempts: int = 3, delay: float = 1.0,
          exceptions: tuple = (Exception,)):
    """Decorator retry với exponential backoff."""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except exceptions as e:
                    if attempt == max_attempts - 1:
                        logger.error(f"{func.__name__} failed sau {max_attempts} lần: {e}")
                        raise
                    wait = delay * (2 ** attempt)
                    logger.warning(f"Attempt {attempt+1}/{max_attempts} failed. Retry sau {wait}s")
                    time.sleep(wait)
        return wrapper
    return decorator

# Sử dụng:
@retry(max_attempts=3, delay=2.0, exceptions=(requests.exceptions.RequestException,))
def fetch_data_from_api(url: str) -> dict:
    response = requests.get(url, timeout=10)
    response.raise_for_status()
    return response.json()

import logging, sys, os

def setup_logging(level: str = "INFO", log_file: str = None):
    handlers = [logging.StreamHandler(sys.stdout)]
    if log_file:
        os.makedirs(os.path.dirname(log_file), exist_ok=True)
        handlers.append(logging.FileHandler(log_file))

    logging.basicConfig(
        level=getattr(logging, level.upper()),
        format="%(asctime)s | %(levelname)-8s | %(name)s | %(message)s",
        datefmt="%Y-%m-%d %H:%M:%S",
        handlers=handlers
    )

import argparse
import pandas as pd
import logging
from datetime import date
from sqlalchemy import create_engine

from config import Config
from extract import read_csv_safe, fetch_all_orders
from transform import clean_orders, create_daily_summary
from load import incremental_load

logger = logging.getLogger(__name__)

class OrderPipeline:
    """Pipeline ETL cho orders data: API → transform → PostgreSQL."""

    def __init__(self, config: Config):
        self.config = config
        self.engine = create_engine(
            f"postgresql+psycopg2://{config.db_user}:{config.db_password}"
            f"@{config.db_host}:{config.db_port}/{config.db_name}"
        )

    def extract(self, run_date: str) -> pd.DataFrame:
        """Extract orders từ API cho ngày chỉ định."""
        logger.info(f"[EXTRACT] Bắt đầu cho ngày {run_date}")
        # Gọi API với pagination
        records = fetch_all_orders(self.config, run_date)
        df = pd.DataFrame(records)
        logger.info(f"[EXTRACT] Xong: {len(df)} rows")
        return df

    def transform(self, df: pd.DataFrame) -> tuple[pd.DataFrame, pd.DataFrame]:
        """Clean và tạo summary. Trả về (clean_orders, daily_summary)."""
        logger.info("[TRANSFORM] Bắt đầu")
        clean = clean_orders(df)
        summary = create_daily_summary(clean)
        logger.info(f"[TRANSFORM] Clean: {len(clean)} rows | Summary: {len(summary)} rows")
        return clean, summary

    def load(self, clean: pd.DataFrame, summary: pd.DataFrame) -> None:
        """Load vào PostgreSQL (incremental)."""
        logger.info("[LOAD] Bắt đầu")
        inserted = incremental_load(clean, self.engine, "orders", "order_id")
        summary.to_sql("daily_summary", self.engine, if_exists="append", index=False)
        logger.info(f"[LOAD] Xong: {inserted} orders mới, {len(summary)} summary rows")

    def run(self, run_date: str) -> None:
        """Chạy toàn bộ pipeline E → T → L."""
        logger.info(f"=== Pipeline bắt đầu: {run_date} ===")
        raw = self.extract(run_date)
        clean, summary = self.transform(raw)
        self.load(clean, summary)
        logger.info(f"=== Pipeline hoàn thành: {run_date} ===")

def parse_args():
    parser = argparse.ArgumentParser(description="Order ETL Pipeline")
    parser.add_argument(
        "--date", type=str,
        default=str(date.today()),
        help="Ngày chạy pipeline (YYYY-MM-DD), mặc định: hôm nay"
    )
    parser.add_argument(
        "--log-level", type=str, default="INFO",
        choices=["DEBUG", "INFO", "WARNING", "ERROR"]
    )
    return parser.parse_args()

if __name__ == "__main__":
    args = parse_args()
    setup_logging(args.log_level, "logs/pipeline.log")

    config = Config()
    config.validate()

    pipeline = OrderPipeline(config)
    pipeline.run(args.date)
