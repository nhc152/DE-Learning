2) 🎯 Mục Đích Module▼

Data Engineering thực tế có câu: "Garbage In, Garbage Out". Pipeline sạch nhưng data bẩn = insight sai = quyết định sai. Theo nhiều khảo sát ngành, data engineer dành 60–80% thời gian cho data cleaning và preparation.

🔴 Hậu quả của data bẩn không được xử lý

  • Business impact: Report sai → CEO ra quyết định dựa trên số liệu lệch → thiệt hại tài chính thực tế
  • Model ML: Dữ liệu training bẩn → model dự báo sai → AML false positive/negative (ngành ngân hàng)
  • Pipeline downstream: JOIN trên key sai định dạng → mất dữ liệu im lặng (silent data loss)
  • SLA: NULL cascade trong aggregation → metric bị undercount → SLA báo sai

🟢 Medallion Architecture — Nơi Cleaning Xảy Ra

Bronze (raw ingestion) → Silver (cleaned + validated) → Gold (business aggregates). Silver layer là nơi áp dụng toàn bộ cleaning + validation trước khi downstream analytics dùng dữ liệu.

🔁 Chu trình vận hành Module 103 (artifact-first)

  • Identify: đo null rate, duplicate rate, outlier rate, wrong type rate trước khi sửa.
  • Strategies: áp dụng imputation/dedup/normalization có giải thích vì sao chọn.
  • Validate: kiểm tra schema + business rules; fail-loudly nếu vi phạm critical rules.
  • Before/After: bắt buộc ghi số liệu cải thiện vào before_after_comparison.md.
💡Banking/AML context: Dữ liệu giao dịch thường có amount=NULL, account_no sai format, duplicate do retry, timestamp lệch timezone. Nếu không profile trước và so sánh sau cleaning, pipeline có thể "đẹp log" nhưng vẫn gây lệch compliance.

-- PostgreSQL: COALESCE + median by group + audit flags
WITH median_by_type AS (
    SELECT
        txn_type,
        PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY amount) AS median_amount
    FROM transactions
    WHERE amount IS NOT NULL
    GROUP BY txn_type
)
SELECT
    t.transaction_id,
    t.txn_type,
    COALESCE(t.amount, m.median_amount) AS amount_clean,
    (t.amount IS NULL)::INT AS is_amount_imputed,
    CASE WHEN t.account_no IS NULL OR TRIM(t.account_no)=''
         THEN 1 ELSE 0 END AS is_rejected_key
FROM transactions t
LEFT JOIN median_by_type m ON t.txn_type = m.txn_type;

-- Oracle 19c: NVL/NVL2 + MEDIAN by group + audit flags
SELECT
    t.transaction_id,
    t.txn_type,
    NVL(
      t.amount,
      MEDIAN(t.amount) OVER (PARTITION BY t.txn_type)
    ) AS amount_clean,
    NVL2(t.amount, 0, 1) AS is_amount_imputed,
    CASE WHEN t.account_no IS NULL
         THEN 1 ELSE 0 END AS is_rejected_key
FROM transactions t;

-- Forward fill cho time series — Oracle 12c+
SELECT account_no, txn_date, balance,
    LAST_VALUE(balance IGNORE NULLS)
      OVER (
        PARTITION BY account_no
        ORDER BY txn_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
      ) AS balance_filled
FROM account_daily_balance;

    SELECT ctid,
        ROW_NUMBER() OVER (
            PARTITION BY transaction_id
            ORDER BY created_at DESC

    FROM transactions
)

-- ROW_NUMBER pattern — giữ row mới nhất

FROM (

        ROW_NUMBER() OVER (
            PARTITION BY transaction_id
            ORDER BY created_at DESC, source_system ASC
        ) AS rn
    FROM transactions t
)
WHERE rn = 1;

-- Hard delete với ROWID (Oracle-efficient)
DELETE FROM transactions
WHERE ROWID NOT IN (
    SELECT MIN(ROWID)
    FROM transactions
    GROUP BY transaction_id
);

-- Oracle-specific: KEEP (DENSE_RANK FIRST) aggregate
SELECT
    transaction_id,
    MIN(amount) KEEP (DENSE_RANK FIRST ORDER BY created_at DESC) AS latest_amount,
    MIN(created_at) KEEP (DENSE_RANK FIRST ORDER BY created_at DESC) AS latest_ts
FROM transactions
GROUP BY transaction_id;

-- IQR Outlier Detection — PostgreSQL
WITH stats AS (
    SELECT
        PERCENTILE_CONT(0.25) WITHIN GROUP (ORDER BY amount) AS q1,
        PERCENTILE_CONT(0.75) WITHIN GROUP (ORDER BY amount) AS q3
    FROM transactions WHERE amount IS NOT NULL
),
bounds AS (
    SELECT q1, q3,
        q1 - 1.5 * (q3 - q1) AS lower_fence,
        q3 + 1.5 * (q3 - q1) AS upper_fence
    FROM stats
)
SELECT t.*,
    CASE WHEN t.amount < b.lower_fence OR t.amount > b.upper_fence
         THEN 'OUTLIER' ELSE 'NORMAL' END AS outlier_flag
FROM transactions t, bounds b;

-- Validate date string format với regex TRƯỚC KHI CAST
SELECT txn_date_raw,
    CASE
        WHEN txn_date_raw ~ '^[0-9]{4}-[0-9]{2}-[0-9]{2}$'
        THEN txn_date_raw::DATE
        ELSE NULL
    END AS txn_date_parsed,
    txn_date_raw !~ '^[0-9]{4}-[0-9]{2}-[0-9]{2}$' AS is_invalid_date
FROM raw_transactions;

        ELSE NULL

FROM raw_transactions;

WITH base_stats AS (
    SELECT
        COUNT(*)                                     AS total_rows,
        COUNT(*) - COUNT(transaction_id)            AS null_txn_id,
        COUNT(DISTINCT transaction_id)               AS unique_txn_id,
        COUNT(*) - COUNT(amount)                   AS null_amount,
        MIN(amount) AS min_amt,  MAX(amount) AS max_amt,
        AVG(amount) AS avg_amt,  STDDEV(amount) AS std_amt,
        PERCENTILE_CONT(0.5) WITHIN GROUP
            (ORDER BY amount)                        AS median_amt,
        MIN(txn_date) AS earliest_date,
        MAX(txn_date) AS latest_date,
        CURRENT_DATE - MAX(txn_date)               AS days_stale,
        COUNT(DISTINCT account_no)                   AS unique_accounts
    FROM transactions
),
dup_check AS (
    SELECT COUNT(*) AS dup_rows
    FROM (SELECT transaction_id
          FROM transactions
          GROUP BY transaction_id HAVING COUNT(*) > 1) d
)
SELECT b.*, d.dup_rows,
    ROUND(null_txn_id * 100.0 / total_rows, 2) AS null_txn_pct,
    ROUND(null_amount * 100.0 / total_rows, 2)  AS null_amt_pct
FROM base_stats b, dup_check d;

transaction_schema = DataFrameSchema({
    "transaction_id": Column(str,
        checks=[Check.str_matches(r'^TXN[0-9]{10}$')],
        nullable=False),
    "account_no": Column(str,
        checks=[Check.str_length(10, 16)],
        nullable=False),
    "amount": Column(float,
        checks=[Check.greater_than_or_equal_to(0),
                Check.less_than_or_equal_to(1_000_000_000)],
        nullable=True),
    "txn_date": Column("datetime64[ns]",
        checks=[Check.less_than_or_equal_to(pd.Timestamp('today'))],
        nullable=False),
    "txn_type": Column(str,
        checks=[Check.isin(["DEBIT", "CREDIT", "TRANSFER"])],
        nullable=False)
}, strict=True)
