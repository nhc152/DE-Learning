3. Giải Thích Chi Tiết ▼

OVER clause gồm 3 phần cốt lõi

  • PARTITION BY: chia dữ liệu thành các nhóm độc lập.
  • ORDER BY: thứ tự tính toán bên trong mỗi partition.
  • Frame: phạm vi hàng tham gia tính toán tại mỗi dòng.
postgresql + oracle sql · running_total.sql
-- Running total + moving average 7 ngày
SELECT
  order_date,
  revenue,
  SUM(revenue) OVER (
    ORDER BY order_date
    ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
  ) AS running_total,
  AVG(revenue) OVER (
    ORDER BY order_date
    ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
  ) AS moving_avg_7d
FROM fct_daily_revenue
ORDER BY order_date;
postgresql + oracle sql · rank_in_category.sql
-- So sánh ROW_NUMBER vs RANK vs DENSE_RANK
SELECT
  category,
  product_id,
  sales,
  ROW_NUMBER() OVER (PARTITION BY category ORDER BY sales DESC) AS rn,
  RANK() OVER (PARTITION BY category ORDER BY sales DESC) AS rnk,
  DENSE_RANK() OVER (PARTITION BY category ORDER BY sales DESC) AS dense_rnk
FROM fct_product_sales;
postgresql + oracle sql · lag_lead_compare.sql
-- So sánh doanh thu với kỳ trước và kỳ sau
SELECT
  month,
  revenue,
  LAG(revenue, 1, 0) OVER (ORDER BY month) AS prev_month_revenue,
  LEAD(revenue, 1, 0) OVER (ORDER BY month) AS next_month_revenue,
  revenue - LAG(revenue, 1, 0) OVER (ORDER BY month) AS delta_prev
FROM fct_monthly_revenue;
postgresql + oracle sql · retention_percentile.sql
-- Phân vị retention của user cohort
SELECT
  cohort_month,
  user_id,
  retention_30d,
  PERCENT_RANK() OVER (
    PARTITION BY cohort_month
    ORDER BY retention_30d
  ) AS retention_percentile
FROM fct_user_retention;
postgresql + oracle sql · top3_each_month.sql
-- Top 3 sản phẩm bán chạy nhất mỗi tháng
WITH ranked AS (
  SELECT
    month,
    product_id,
    sales,
    ROW_NUMBER() OVER (
      PARTITION BY month
      ORDER BY sales DESC
    ) AS rn
  FROM fct_product_monthly_sales
)
SELECT month, product_id, sales
FROM ranked
WHERE rn <= 3
ORDER BY month, rn;

Bài 1 · Running total + moving avg

Với bảng fct_daily_revenue(day, revenue), viết query trả về running total và moving average 7 ngày.

solution.sql
SELECT
  day,
  revenue,
  SUM(revenue) OVER (ORDER BY day ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) AS running_total,
  AVG(revenue) OVER (ORDER BY day ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) AS moving_avg_7d
FROM fct_daily_revenue;
Bài 2 · Top 3 sản phẩm mỗi category

Viết query lấy top 3 sản phẩm có doanh số cao nhất trong từng category.

solution.sql
WITH ranked AS (
  SELECT
    category,
    product_id,
    sales,
    ROW_NUMBER() OVER (PARTITION BY category ORDER BY sales DESC) AS rn
  FROM fct_product_sales
)
SELECT *
FROM ranked
WHERE rn <= 3;
Bài 3 · LAG và tỷ lệ tăng trưởng

Từ bảng doanh thu tháng, tính tăng trưởng % so với tháng trước cho từng sản phẩm.

solution.sql
SELECT
  product_id,
  month,
  revenue,
  LAG(revenue) OVER (PARTITION BY product_id ORDER BY month) AS prev_revenue,
  CASE WHEN LAG(revenue) OVER (PARTITION BY product_id ORDER BY month) = 0 THEN NULL
       ELSE (revenue - LAG(revenue) OVER (PARTITION BY product_id ORDER BY month))
            * 100.0 / LAG(revenue) OVER (PARTITION BY product_id ORDER BY month)
  END AS growth_pct
FROM fct_product_monthly_revenue;

Vấn đề của cách làm truyền thống

Nếu cần tổng hợp theo nhiều chiều như (store, category), (store), (category), (all), nhiều người viết 4 query UNION ALL. Cách đó dài, khó bảo trì và dễ sai khi thêm metric mới.

postgresql + oracle sql · grouping_sets.sql
-- Sales report đa chiều gọn bằng GROUPING SETS
SELECT
  store_id,
  category,
  SUM(sales_amount) AS total_sales,
  GROUPING(store_id) AS g_store,
  GROUPING(category) AS g_category
FROM fct_sales
GROUP BY GROUPING SETS (
  (store_id, category),
  (store_id),
  (category),
  ()
)
ORDER BY store_id NULLS LAST, category NULLS LAST;
postgresql + oracle sql · rollup_sales.sql
-- Subtotal theo hierarchy thời gian
SELECT
  year,
  quarter,
  month,
  SUM(sales_amount) AS total_sales
FROM fct_sales_calendar
GROUP BY ROLLUP (year, quarter, month)
ORDER BY year, quarter, month;
postgresql + oracle sql · cube_activity.sql
-- Mọi tổ hợp nền tảng và action
SELECT
  platform,
  action_type,
  COUNT(*) AS events
FROM fct_user_events
GROUP BY CUBE (platform, action_type)
ORDER BY platform NULLS LAST, action_type NULLS LAST;

Bài 2 · Label subtotal với GROUPING()

Trong output của bài 1, thêm cột level_label để phân biệt detail, subtotal, grand total.

solution.sql
CASE
  WHEN GROUPING(store_id)=0 AND GROUPING(category)=0 THEN 'detail'
  WHEN GROUPING(store_id)=0 AND GROUPING(category)=1 THEN 'store subtotal'
  WHEN GROUPING(store_id)=1 AND GROUPING(category)=0 THEN 'category subtotal'
  ELSE 'grand total'
END AS level_label

3. Giải Thích Chi Tiết ▼

SCD Type 1: overwrite nhanh, không giữ lịch sử

📌 Phần dưới hiển thị song song 2 dialect: PostgreSQL và Oracle SQL để bạn dùng theo hệ quản trị đang học.

PostgreSQL · SCD Type 1

postgresql · scd_type1.sql
-- Cập nhật trực tiếp bản ghi hiện tại
UPDATE dim_customer d
SET
  city = s.city,
  email = s.email,
  updated_at = CURRENT_TIMESTAMP
FROM stg_customer s
WHERE d.customer_id = s.customer_id

PostgreSQL · Incremental watermark

postgresql · incremental_timestamp.sql
-- Lấy watermark cuối lần load trước
WITH wm AS (
  SELECT COALESCE(MAX(updated_at), '1970-01-01'::timestamp) AS last_wm
  FROM dw_orders
)
INSERT INTO dw_orders (order_id, customer_id, amount, updated_at)
SELECT s.order_id, s.customer_id, s.amount, s.updated_at
FROM src_orders s
CROSS JOIN wm
WHERE s.updated_at > wm.last_wm;

Idempotent UPSERT

PostgreSQL · ON CONFLICT

postgresql · incremental_upsert.sql
-- Tránh duplicate khi chạy lại
INSERT INTO dw_orders (order_id, customer_id, amount, updated_at)
SELECT order_id, customer_id, amount, updated_at
FROM stg_orders_delta
ON CONFLICT (order_id)
DO UPDATE SET
  customer_id = EXCLUDED.customer_id,
  amount = EXCLUDED.amount,
  updated_at = EXCLUDED.updated_at
WHERE dw_orders.updated_at < EXCLUDED.updated_at;

Oracle SQL · MERGE

oracle sql · incremental_upsert.sql
-- Tránh duplicate khi chạy lại bằng MERGE
MERGE INTO dw_orders d
USING stg_orders_delta s
   ON (d.order_id = s.order_id)
 WHEN MATCHED THEN
   UPDATE SET
     d.customer_id = s.customer_id,
     d.amount = s.amount,
     d.updated_at = s.updated_at
   WHERE d.updated_at < s.updated_at
 WHEN NOT MATCHED THEN
   INSERT (order_id, customer_id, amount, updated_at)
   VALUES (s.order_id, s.customer_id, s.amount, s.updated_at);
